<script>
const decoracionesPrices = {
    liso_strass: 800,
    relieve_1: 100,
    relieve_2: 300,
    aura: 400,
    french: 200,
    nailart: 250,
    strass: 250,
    cateye: 200,
    bijou: 200,
    aniadido: 1000
};

const nombresDecoraciones = {
    liso_strass: "Liso completo strass",
    relieve_1: "Relieve",
    relieve_2: "Relieve Especial",
    aura: "Aura",
    french: "French",
    nailart: "Nail Art",
    strass: "Strass",
    cateye: "Cat Eye",
    bijou: "Bijouterie",
    aniadido: "Añadido de uña"
};

document.addEventListener('DOMContentLoaded', () => {
    const container = document.getElementById('decoraciones-container');

    for (let key in decoracionesPrices) {
        const div = document.createElement('div');
        div.className = "flex justify-between items-center";

        div.innerHTML = `
            <span>${nombresDecoraciones[key]}</span>
            <div class="flex items-center gap-2">
                <button onclick="changeQty('${key}', -1)">-</button>
                <input id="${key}" value="0" class="w-10 text-center border rounded">
                <button onclick="changeQty('${key}', 1)">+</button>
            </div>
        `;

        container.appendChild(div);
    }

    calculateTotal();
});

function changeQty(id, delta) {
    const input = document.getElementById(id);
    let val = parseInt(input.value) + delta;
    if (val < 0) val = 0;
    if (val > 10) val = 10;
    input.value = val;
    calculateTotal();
}

function calculateTotal() {
    let total = 0;

    const tecnica = document.querySelector('input[name="tecnica"]:checked');
    total += parseInt(tecnica.value);

    const retiro = document.querySelector('input[name="retiro"]:checked');
    total += parseInt(retiro.value);

    for (let key in decoracionesPrices) {
        const qty = parseInt(document.getElementById(key).value);
        if (qty > 0) {
            total += qty * decoracionesPrices[key];
        }
    }

    document.getElementById('total-display').innerText =
        '$' + total.toLocaleString('es-AR');

    return total;
}

function shareOnWhatsApp() {
    const total = calculateTotal();
    let msg = `Presupuesto Oh My Nails 💅\nTotal: $${total}`;
    window.open(`https://wa.me/?text=${encodeURIComponent(msg)}`);
}
